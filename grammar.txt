
TERMINAL_SYMBOLS = [NEWLINE, EOF, COMMA, ID, OPERATOR, SEMICOLON, 
                PROPERTY_NAME, IMPORTANT, TEXT_VALUE, HEX_VALUE,
                REM_VAlUE, PERC_VALUE, URL_VALUE, INT_VALUE, NAME_VALUE, PX_VALUE]

NIETER


START -> SEPARATION ENTRY EOF
ENTRY -> ENTRY ENTRY
ENTRY -> SELECT NEWLINE INDENT PROPERTIES DEDENT SEPARATION

SEPARATION -> ε 
SEPARATION -> NEWLINE SEPARATION 

SELECT -> SELECT COMMA NEWLINE SELECT
SELECT -> LIST_OF_IDS
SELECT -> ID OPERATOR ID
LIST_OF_IDS -> ID LIST_OF_IDS
LIST_OF_IDS -> ε

# popracuj nad SEMICOLON bo teraz jest zle (chyba jest dobrze)
PROPERTIES -> PROPERTIES SEMICOLON NEWLINE PROPERTY 
PROPERTIES - > PROPERTY
PROPERTY -> PROPERTY_NAME VALUE


VALUE -> VALUE IMPORTANT
VALUE -> TEXT_VALUE | HEX_VALUE |REM_VAlUE | PERC_VALUE | URL_VALUE | INT_VALUE | ID | PX_VALUE 
VALUE -> PX_VALUE ID HEX_VALUE
